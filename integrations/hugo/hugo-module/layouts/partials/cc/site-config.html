{{- /* Serializes the browser-relevant slice of the site configuration. The runtime writes this (plus editor overrides) as the WASM site's config, so component partials can use `site.Params`, `site.Title`, and `site.Menus`. */ -}} {{- $menus := dict -}} {{- range $name, $entries := site.Menus -}} {{- $items := slice -}} {{- range $entries -}} {{- $items = $items | append (dict "name" .Name "url" .URL "weight" .Weight) -}} {{- end -}} {{- $menus = merge $menus (dict $name $items) -}} {{- end -}} {{- return (dict "baseURL" ("/" | absURL) "title" site.Title "languageCode" (site.LanguageCode | default "") "params" site.Params "menus" $menus ) -}}
