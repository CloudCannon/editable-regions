{{- /* Walks a list of directories, returning one merged dict of path -> file contents. Expects: (dict "dirs" (slice "layouts/partials") "exts" (slice ".html")) */ -}} {{- $exts := .exts -}} {{- $files := dict -}} {{- range $dir := .dirs -}} {{- $files = merge $files (partial "cc/walk-dir.html" (dict "dir" $dir "exts" $exts)) -}} {{- end -}} {{- return $files -}}
